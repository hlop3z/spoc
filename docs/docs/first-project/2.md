# -*- coding: utf-8 -*-
"""{ Core } Read The Docs"""

import spoc

PLUGINS = ["commands", "models", "views"]

@spoc.singleton
class MyFramework:
    """Framework"""

    def init(
        self,
    ):
        """Class __init__ Replacement"""
        framework = spoc.App(plugins=PLUGINS)

        # Collect All Commands
        commands = {}
        for method in framework.component.commands.values():
            commands[method.key] = method.object

        self.commands = commands
        # Use "X" Command
        # commands['demo.hello_world']()
